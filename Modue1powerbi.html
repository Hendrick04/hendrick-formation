function goTo(n) {

  /* Sauvegarder l'étape qu'on quitte si c'est un cours */

  var current = getCurrentStep();

  if (current !== null && isCours(current)) {

    saveEtape(current, "cours_vu", 0, 0);

  }

 

  var sections = document.querySelectorAll('.section');

  for (var i = 0; i < sections.length; i++) sections[i].classList.remove('active');

  document.getElementById('sec' + n).classList.add('active');

  for (var j = 0; j < 7; j++) {

    var d = document.getElementById('dot' + j);

    d.classList.remove('active', 'done');

    if (j < n) d.classList.add('done');

    if (j === n) d.classList.add('active');

  }

  document.getElementById('pgBar').style.width = Math.round(n / 6 * 100) + '%';

  window.scrollTo(0, 0);

}

 

function getCurrentStep() {

  for (var i = 0; i <= 6; i++) {

    var s = document.getElementById('sec' + i);

    if (s && s.classList.contains('active')) return i;

  }

  return null;

}

/* ════════════════════════════════════════

   APPEL APPS SCRIPT (compatible iframe)

════════════════════════════════════════ */

function callSheets(payload, callback) {

  fetch(APPS_SCRIPT_URL, {

    method:  "POST",

    headers: { "Content-Type": "text/plain;charset=UTF-8" },

    body:    JSON.stringify(payload)

  })

  .then(function(r) { return r.text(); })

  .then(function(t) {

    try { callback(JSON.parse(t)); }

    catch(e) { callback({ ok: false }); }

  })

  .catch(function() {

    /* Fallback GET */

    var url = APPS_SCRIPT_URL + "?data=" + encodeURIComponent(JSON.stringify(payload));

    fetch(url)

      .then(function(r) { return r.text(); })

      .then(function(t) { try { callback(JSON.parse(t)); } catch(e) { callback({ ok: false }); } })

      .catch(function() { callback({ ok: false }); });

  });

}

 

/* ═══════════════════════════════════════════

   ENVOI VERS GOOGLE SHEETS

   text/plain évite le preflight CORS dans iframe

═══════════════════════════════════════════ */

function sendToSheets(payload, callback) {

  payload.action = "save_progress";

  fetch(APPS_SCRIPT_URL, {

    method:  "POST",

    headers: { "Content-Type": "text/plain;charset=UTF-8" },

    body:    JSON.stringify(payload)

  })

  .then(function(r) { return r.text(); })

  .then(function(t) {

    try { var r = JSON.parse(t); callback(r.ok === true); }

    catch(e) { callback(true); } /* si parsing échoue mais requête ok */

  })

  .catch(function() {

    /* Fallback : GET avec data en paramètre URL */

    var url = APPS_SCRIPT_URL + "?data=" + encodeURIComponent(JSON.stringify(payload));

    fetch(url)

      .then(function() { callback(true); })

      .catch(function() { callback(false); });

  });

}

/* ═══════════════════════════════════════════

   TÉLÉCHARGEMENT CERTIFICAT PDF

═══════════════════════════════════════════ */

function downloadCertificate() {

  var btn = document.getElementById("dl-cert-btn");

  if (btn) { btn.classList.add("saving"); btn.innerHTML = '<i class="fa-solid fa-circle-notch fa-spin"></i> Génération…'; }

 

  var cert = document.getElementById("cert-block");

  if (!cert) return;

 

  html2canvas(cert, {

    scale: 3,

    useCORS: true,

    backgroundColor: "#ffffff",

    logging: false

  }).then(function(canvas) {

    var imgData = canvas.toDataURL("image/png");

    var pdf = new window.jspdf.jsPDF({

      orientation: "landscape",

      unit: "mm",

      format: "a4"

    });

    var pw = pdf.internal.pageSize.getWidth();

    var ph = pdf.internal.pageSize.getHeight();

    var ratio = canvas.width / canvas.height;

    var imgW = pw - 20;

    var imgH = imgW / ratio;

    if (imgH > ph - 20) { imgH = ph - 20; imgW = imgH * ratio; }

    var x = (pw - imgW) / 2;

    var y = (ph - imgH) / 2;

    pdf.addImage(imgData, "PNG", x, y, imgW, imgH);

    var recipientName = (currentUser.name || "apprenant").replace(/\s+/g, "_");

    pdf.save("Certificat_Module1_" + recipientName + ".pdf");

    if (btn) {

      btn.classList.remove("saving");

      btn.innerHTML = '<i class="fa-solid fa-check"></i> Certificat téléchargé !';

      setTimeout(function() {

        btn.innerHTML = '<i class="fa-solid fa-download"></i> Télécharger le certificat (PDF)';

      }, 3000);

    }

  }).catch(function(err) {

    if (btn) { btn.classList.remove("saving"); btn.innerHTML = '<i class="fa-solid fa-download"></i> Télécharger le certificat (PDF)'; }

    alert("Erreur lors de la génération du PDF. Essaie de faire une capture d\'écran.");

  });

}
